```c++
class BALProblem
{
public:

    bool LoadFile(const std::string& filename);

    int num_cameras_;
    int num_points_;
    int num_observations_;

    std::vector<int> camera_index_;
    std::vector<int> point_index_;

    std::vector<double> observations_;

    std::vector<double> cameras_;
    std::vector<double> points_;

    double* mutable_camera(int i)
    {
        return cameras_.data()+i*9;
    }

    double* mutable_point(int i)
    {
        return points_.data()+i*3;
    }
};
```

```c++
bool BALProblem::LoadFile(const std::string& filename)
{
    std::ifstream file(filename);

    file >> num_cameras_;
    file >> num_points_;
    file >> num_observations_;

    camera_index_.resize(num_observations_);
    point_index_.resize(num_observations_);
    observations_.resize(num_observations_*2);

    for(int i=0;i<num_observations_;i++)
    {
        file >> camera_index_[i];
        file >> point_index_[i];
        file >> observations_[2*i];
        file >> observations_[2*i+1];
    }

    cameras_.resize(num_cameras_*9);
    points_.resize(num_points_*3);

    for(auto& c:cameras_)
        file>>c;

    for(auto& p:points_)
        file>>p;

    return true;
}
```

```c++
void SolveBA(BALProblem& bal)
{
    ceres::Problem problem;

    for(int i=0;i<bal.num_observations_;i++)
    {
        double x = bal.observations_[2*i];
        double y = bal.observations_[2*i+1];

        ceres::CostFunction* cost =
            new ceres::AutoDiffCostFunction
            <ReprojectionError,2,9,3>(
                new ReprojectionError(x,y));

        problem.AddResidualBlock(
            cost,
            nullptr,
            bal.mutable_camera(bal.camera_index_[i]),
            bal.mutable_point(bal.point_index_[i])
        );
    }

    ceres::Solver::Options options;
    options.linear_solver_type=ceres::DENSE_SCHUR;

    ceres::Solver::Summary summary;

    ceres::Solve(options,&problem,&summary);

    std::cout<<summary.FullReport()<<std::endl;
}
```
